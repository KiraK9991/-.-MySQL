236-- запрос фильмов с режиссёрами
237SELECT
238    film_crew.film_id,
239    films.title,
240    films.release_date,
241    films.rating,
242    persons.*
243FROM
244    films LEFT JOIN film_crew
245        ON film_crew.film_id = films.id
246    INNER JOIN persons
247        ON film_crew.person_id = persons.id
248WHERE
249    film_crew.post_id = 2
250ORDER BY
251    persons.name;
252
253-- выбрать только сериалы
254SELECT
255    films.id,
256    films.title,
257    films.short_description,
258    films.release_date,
259    films.rating,
260    films.rating_count,
261    films.is_4k,
262    countries.name,
263    age_ratings.title age_rating,
264    channels.title channel,
265    counts.episode_count
266FROM
267    films,
268    countries,
269    age_ratings,
270    channels,
271    (SELECT film_id, COUNT(*) episode_count FROM episodes GROUP BY film_id) counts
272WHERE
273    films.country_id = countries.id
274    AND films.age_rating_id = age_ratings.id
275    AND films.channel_id = channels.id
276    AND films.id = counts.film_id
277    AND counts.episode_count > 1
278ORDER BY
279    films.title;

329CREATE PROCEDURE add_new_language(IN language VARCHAR(250), IN film_id INT)
330BEGIN
331    DECLARE lang_id INT;
332
333    INSERT INTO languages(title) VALUES (language);

336END//

338-- обновить продолжительность сериала
339delimiter //
340CREATE PROCEDURE update_duration(IN series_id INT)
341BEGIN
342    DECLARE duration_sum INT;
343
344    SELECT SUM(duration) INTO duration_sum FROM episodes WHERE film_id = series_id;
345    UPDATE films SET duration = duration_sum WHERE id = series_id;
346END//
